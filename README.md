<h2>Settings</h2>
<table>
	<tr>
		<th>Option</th>
		<th>Type</th>
		<th>Default</th>
		<th>Description</th>
	</tr>
	<tr>
		<td>autoslide</td>
		<td>boolean</td>
		<td>false</td>
		<td>Enables auto play of slides</td>
	</tr>
	<tr>
		<td>size</td>
		<td>int</td>
		<td>3</td>
		<td>Set the max number of images</td>
	</tr>
	<tr>
		<td>scrollDirection</td>
		<td>string</td>
		<td>horizontal</td>
		<td>Set horizontal or vertical</td>
	</tr>
	<tr>
		<td>imgSize</td>
		<td>int</td>
		<td>50</td>
		<td>If horizontal set img width <br> If vertical set img height</td>
	</tr>
	<tr>
		<td>bigImgClass</td>
		<td>string</td>
		<td>false</td>
		<td>Set css class where to show big img.</td>
	</tr>
</table>
